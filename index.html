// 核心修复：确保异步保存完成后再操作 UI async function saveEdit() { const item = fragments[currentCategory][editingIdx]; // 获取输入值 const newName = document.getElementById('editName').value.trim(); const newValue = document.getElementById('editValue').value.trim(); if(!newName || !newValue) { alert("名称和内容不能为空"); return; } // 更新内存中的对象 item.name = newName; item.value = newValue; const file = document.getElementById('editImgInput').files[0]; if(file) { // 如果有新图片，等待转换完成 item.img = await new Promise((resolve) => { const rd = new FileReader(); rd.readAsDataURL(file); rd.onload = (e) => resolve(e.target.result); }); } try { // 关键点：使用 await 等待数据库写入操作彻底完成 await saveData(); // 只有保存成功后才关闭弹窗并更新显示 closeModal(); renderAll(); alert("✅ 修改已永久保存"); } catch (err) { console.error("保存失败:", err); alert("❌ 数据库写入失败，请检查手机存储空间。"); } }
